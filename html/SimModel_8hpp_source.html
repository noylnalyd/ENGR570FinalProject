273  double SimModel::skinHeat(

288  double f = 1;
289  double echelonSf = body->elements[elemIdx]->sectors[sectIdx]->echelon;
290  double echelonSr = echelonSrm;

304  elem->a_nat*sqrt(abs(TsfC-TairC))

315 
316

318  double Posksat = 100.0*exp(18.956-4030.0/(TsfC+235));
